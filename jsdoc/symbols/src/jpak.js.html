 44 /**
 45  * Clean all deletedValue from array
 46  * @expose
 47  */
 48 Array.prototype.clean = function(deleteValue) {
 49   for (var i = 0; i < this.length; i++) {
 50     if (this[i] == deleteValue) {         
 51       this.splice(i, 1);
 52       i--;
 53     }
 54   }
 55   return this;
 56 };

 67  * Convert Unsigned Int8 ArrayBuffer to String
 68  * @param {Uint8Array} uintArray
 69  * @return {string}
 70  */
 71 JPAK.Uint8ArrayToString = function(uintArray) {
 72     var o = "";
 73     for(var i=0;i<uintArray.byteLength;i++)  
 74         o += String.fromCharCode(uintArray[i]);
 75     return o;
 76 }

 84 /**
 85  * Convert a String to an ArrayBuffer using uint8
 86  * @expose
 87  * @param {string} str
 88  * @return {ArrayBuffer}
 89  */
 90 JPAK.String2ArrayBuffer = function(str)   {
 91     var buf = new ArrayBuffer(str.length);
 92     var bufView = new Uint8Array(buf);
 93     for (var i=0, strLen=str.length; i<strLen; i++) 
 94         bufView[i] = str.charCodeAt(i) & 0xFF;
 95     return buf;
 96 };

 98 /**
 99  * Returns a Base64 String from an ArrayBuffer
100  * Modified version from https://gist.github.com/jonleighton/958841
101  * @expose
102  * @param {arrayBuffer} arrayBuffer
103  * @return {string} base64
104  */
105 JPAK.ArrayBufferToBase64 = function(arrayBuffer)  {
106   var base64    = ''
107 
108   var bytes         = new Uint8Array(arrayBuffer)
109   var byteLength    = bytes.byteLength
110   var byteRemainder = byteLength % 3
111   var mainLength    = byteLength - byteRemainder
112 
113   var a, b, c, d
114   var chunk
115 
116   // Main loop deals with bytes in chunks of 3
117   for (var i = 0; i < mainLength; i = i + 3) {
118     // Combine the three bytes into a single integer
119     chunk = (bytes[i] << 16) | (bytes[i + 1] << 8) | bytes[i + 2]
120 
121     // Use bitmasks to extract 6-bit segments from the triplet
122     a = (chunk & 16515072) >> 18 // 16515072 = (2^6 - 1) << 18
123     b = (chunk & 258048)   >> 12 // 258048   = (2^6 - 1) << 12
124     c = (chunk & 4032)     >>  6 // 4032     = (2^6 - 1) << 6
125     d = chunk & 63               // 63       = 2^6 - 1
126 
127     // Convert the raw binary segments to the appropriate ASCII encoding
128     base64 += JPAK.Base64_Encoding[a] + JPAK.Base64_Encoding[b] + JPAK.Base64_Encoding[c] + JPAK.Base64_Encoding[d]
129   }
130 
131   // Deal with the remaining bytes and padding
132   if (byteRemainder == 1) {
133     chunk = bytes[mainLength]
134 
135     a = (chunk & 252) >> 2 // 252 = (2^6 - 1) << 2
136 
137     // Set the 4 least significant bits to zero
138     b = (chunk & 3)   << 4 // 3   = 2^2 - 1
139 
140     base64 += JPAK.Base64_Encoding[a] + JPAK.Base64_Encoding[b] + '=='
141   } else if (byteRemainder == 2) {
142     chunk = (bytes[mainLength] << 8) | bytes[mainLength + 1]
143 
144     a = (chunk & 64512) >> 10 // 64512 = (2^6 - 1) << 10
145     b = (chunk & 1008)  >>  4 // 1008  = (2^6 - 1) << 4
146 
147     // Set the 2 least significant bits to zero
148     c = (chunk & 15)    <<  2 // 15    = 2^4 - 1
149 
150     base64 += JPAK.Base64_Encoding[a] + JPAK.Base64_Encoding[b] + JPAK.Base64_Encoding[c] + '='
151   }
152 
153   return base64
154 };

180 /**
181  * Searches for a file on the cache
182  * Returns undefined if not found
183  * @param {string} path
184  * @return {Object} file
185  */
186 JPAK.jpakloader.prototype.CacheLoad     =   function(path)  {
187     for(var i=0;i<this.filecache.length;i++)    {
188         if(this.filecache[i].path == path)
189             return this.filecache[i];
190     }
191     return undefined;
192 };

194 //  Loads the jpak file and process it
195 /**
196  * Loads the JPAK File and Process it
197  */
198 JPAK.jpakloader.prototype.Load = function() {
199     if(this.jpakfile !== undefined) {
200         // _this is used to reference the jpakloader object
201         var _this = this;
202         
203         //  Lets create a new XMLHttpRequest to load the jpak
204         var xhr = new XMLHttpRequest();
205         
206         xhr.open('GET', this.jpakfile, true);
207         xhr.responseType = 'arraybuffer';       //  We want an ArrayBuffer for processing
208         
209         //  The On Progress request. For now it only does stuff if there is a hooked onprogress jpakloader event.
210         xhr.onprogress = function(e)    {
211             if (e.lengthComputable && _this.onprogress != undefined)     {  
212                 var percentComplete = (( (e.loaded / e.total)*10000 ) >> 0)/100;  // Rounded percent to two decimal
213                     _this.onprogress({"loaded":e.loaded,"total":e.total,"percent": percentComplete});                
214             }   
215         }
216         
217         //  The onload function. This parses the JPAK and after loading the filetable it calls the onload event of jpakloader
218         xhr.onload = function(e) {
219             if (this.status == 200) {
220                 var data = this.response;
221                 var MagicNumber = u8as(new Uint8Array(data.slice(0,5)));
222                 if(MagicNumber == "JPAK1")  {
223                     JPAK.log("JPAK::jpakloader - Loaded file "+_this.jpakfile+" successfully. JPAK1 Format");
224                     var filetableoffset = new DataView(data.slice(data.byteLength-4,data.byteLength)).getUint32(0, true);
225                     var filetable = new Uint8Array(data.slice(filetableoffset,data.byteLength-4));
226                     filetable = JSON.parse(u8as(filetable));
227                     _this.filetable = filetable;
228                     _this.jpakdata = data;
229                     _this.dataloaded = true;
230                     if(_this.onload != undefined)   
231                         _this.onload();
232                 }else{
233                     JPAK.log("JPAK::jpakloader - Error loading file "+_this.jpakfile+" (8000): Wrong File Magic. Expected JPAK1 got "+MagicNumber);
234                     if(_this.onerror != undefined)
235                         _this.onerror({"text": "Wrong File Magic. Expected JPAK1 got "+MagicNumber, "errorcode" : 8000}); 
236                 }
237             }
238         };
239         xhr.onreadystatechange = function (aEvt) {
240             if (this.readyState == 4) {
241                 if(this.status != 200)   {
242                     JPAK.log("JPAK::jpakloader - Error loading file "+_this.jpakfile+" ("+this.status+"): "+this.statusText);
243                     if(_this.onerror != undefined)
244                         _this.onerror({"text": this.statusText, "errorcode": this.status});
245                 }
246           }
247         };
248         //  Send the request
249         xhr.send();  
250     }else
251         console.log("JPAK::jpakloader - No file to load!");
252 };

254 /**
255  * Gets the directory entry if exists.
256  * Returns undefined if not found
257  * @param {string} path
258  * @return {object} directoryentry
259  */
260 JPAK.jpakloader.prototype.FindDirectoryEntry = function(path)   {
261     var base = this.filetable;
262     if(this.dataloaded) {
263         if(path != "/") {
264             path = path.split("/").clean("");
265             var dir = "", ok = true;
266             for(var i=0;i<path.length;i++)    {
267                 dir = path[i];
268                 if(dir in base.directories)  
269                     base = base.directories[dir]; 
270                 else{
271                     ok = false;
272                     break;
273                 }
274             }
275             if(!ok)
276                 base = undefined;        
277         }
278     }
279     return base;
280 };

282 /**
283  * Gets the file entry if exists.
284  * Returns undefined if not found
285  * @param {string} path
286  * @return {object} fileentry
287  */
288 JPAK.jpakloader.prototype.FindFileEntry = function(path)    {
289     var pathblock = path.split("/").clean("");
290     var filename  = pathblock[pathblock.length-1];
291     path = path.replace(filename,"");
292     var base = this.FindDirectoryEntry(path);
293     if(base != undefined)   
294         if(filename in base.files)  
295             return base.files[filename];
296     return undefined;
297 };

299 /**
300  * Lists the dir returning an object like:
301  * { "dirs" : [ arrayofdirs ], "files" : [ arrayoffiles ], "error" : "An error message, if happens" }
302  * @param {string} path
303  * @return {object} dirlist
304  */
305 JPAK.jpakloader.prototype.ls = function(path)   {
306     var out = { "files" : [], "dirs" : [] };
307     if(this.dataloaded) {
308         var base = this.FindDirectoryEntry(path);
309         if(base != undefined)  {
310             for(var i in base.files)
311                 out.files.push(base.files[i]);
312             for(var i in base.directories)
313                 out.dirs.push({"name" : base.directories[i].name, "numfiles": base.directories[i].numfiles});
314         }else
315             out.error = "Directory not found!";
316                
317     }else
318         out.error = "Not loaded";   
319     return out;
320 };

322 /**
323  * Returns a blob of the file. 
324  * It looks in the cache for already loaded files.
325  * @param {path} File Path
326  * @param {type} File Mime Type
327  * @return {Blob} File Blobs 
328  */
329 JPAK.jpakloader.prototype.GetFile = function(path, type)  {
330     var file = this.FindFileEntry(path);
331     type = type || 'application/octet-binary';
332     var cache = this.CacheLoad(path);
333     
334     if(file != undefined && cache == undefined)  { 
335         //  Add it to file cache
336         var dataslice = this.jpakdata.slice(file.offset,file.offset+file.size);
337         if(file.compressed !== undefined && file.compressed)
338             dataslice = JPAK.GZIP.decompress(dataslice);
339         var blob = new Blob([new Uint8Array(dataslice).buffer], { "type":type});
340         this.filecache.push({"path":path,"type":type,"blob":blob,"url":URL.createObjectURL(blob), "arraybuffer" : dataslice} );
341         return blob;
342     }else if(cache != undefined)
343         return cache.blob;
344         
345     return undefined;
346 };

362             JPAK.log("Error: Cannot find file \""+path+"\"");

365     }else
366         return cache.url;

397 /**
398  * Returns an Base64 Encoded File Content.
399  * It looks in the cache for already loaded files.
400  * @param {path} File Path
401  * @param {type} File Mime Type
402  * @return {string} Base64 String
403  */
404 JPAK.jpakloader.prototype.GetBase64File = function(path, type) {
405     var filedata = this.GetFileArrayBuffer(path, type);
406     if(filedata == undefined)
407         return undefined;
408     
409     return JPAK.ArrayBufferToBase64(filedata);

412 /**
413  * Returns an HTML Data URI with File Content.
414  * It looks in the cache for already loaded files.
415  * Using HTML Data URI for Images, you can hide the load process from chrome Network Inspector
416  * I didnt find any place that you can find DataURI File
417  * @param {path} File Path
418  * @param {type} File Mime Type
419  * @return {string} HTML Data URI
420  */
421 JPAK.jpakloader.prototype.GetHTMLDataURIFile = function(path, type, encoding) {
422     var b64 = this.GetBase64File(path, type);
423     // HTML Data URI Format: data:[<MIME-type>][;charset=<encoding>][;base64],<data>
424     if(b64 === undefined)
425         return undefined
426         
427     if(encoding !== undefined)
428         return "data:"+type+";charset="+encoding+";base64,"+b64;
429     else
430         return "data:"+type+";base64,"+b64;
431 };
